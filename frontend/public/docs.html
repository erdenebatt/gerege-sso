1. Тойм

Gerege SSO нь Монголын иргэдийн дижитал таних системээр аппликейшнүүдэд нэвтрэх боломжийг олгодог OAuth2 зөвшөөрлийн сервер юм.

Онцлогууд

  • OAuth2 Authorization Code Grant (+ PKCE дэмжлэг)
  • Google, Apple, Facebook, Twitter-ээр нэвтрэх
  • Регистрийн дугаараар иргэний баталгаажуулалт
  • JWT токен дотор баяжуулсан иргэний мэдээлэл
Чухал: Регистрийн дугаар (reg_no) болон иргэний дугаар (civil_id) нь гуравдагч талын токенд хэзээ ч орохгүй. Зөвхөн gen_id (11 оронтой өвөрмөц дугаар) дамжуулагдана.

2. Хурдан эхлэл

Клиент бүртгүүлэх

Gerege SSO админтай холбогдож дараах мэдээллийг өгнө үү:

  • Аппликейшны нэр (зөвшөөрлийн дэлгэцэнд харагдана)
  • Redirect URI (зөвшөөрлийн дараа буцах хаяг)

Та дараах мэдээллийг хүлээн авна:

  • client_id — Нийтийн танигч
  • client_secret — Нууц түлхүүр (backend-д хадгална, хэзээ ч frontend-д бичихгүй)

4. PKCE хамгаалалт

PKCE (Proof Key for Code Exchange) нь authorization code interception халдлагаас хамгаалдаг. Mobile болон SPA аппликейшнүүдэд зайлшгүй шаардлагатай.

Хэрхэн ажилладаг

  1. Санамсаргүй code_verifier үүсгэнэ (43-128 тэмдэгт)
  2. SHA-256 hash хийж code_challenge үүсгэнэ
  3. Authorize хүсэлтэд code_challenge илгээнэ
  4. Token хүсэлтэд code_verifier илгээнэ
// JavaScript PKCE implementation async function generatePKCE() { // Generate code_verifier (43-128 characters) const array = new Uint8Array(32); crypto.getRandomValues(array); const codeVerifier = btoa(String.fromCharCode(...array)) .replace(/\+/g, '-') .replace(/\//g, '_') .replace(/=/g, ''); // Generate code_challenge (S256) const encoder = new TextEncoder(); const data = encoder.encode(codeVerifier); const hash = await crypto.subtle.digest('SHA-256', data); const codeChallenge = btoa(String.fromCharCode(...new Uint8Array(hash))) .replace(/\+/g, '-') .replace(/\//g, '_') .replace(/=/g, ''); return { codeVerifier, codeChallenge }; } // Usage const { codeVerifier, codeChallenge } = await generatePKCE(); sessionStorage.setItem('pkce_verifier', codeVerifier); // Authorize URL const authUrl = `https://sso.gerege.mn/api/oauth/authorize?` + `client_id=${CLIENT_ID}&` + `redirect_uri=${encodeURIComponent(REDIRECT_URI)}&` + `response_type=code&` + `scope=openid+profile&` + `code_challenge=${codeChallenge}&` + `code_challenge_method=S256&` + `state=${generateState()}`; window.location.href = authUrl;
# Python PKCE implementation import secrets import hashlib import base64 def generate_pkce(): # Generate code_verifier code_verifier = secrets.token_urlsafe(32) # Generate code_challenge (S256) digest = hashlib.sha256(code_verifier.encode()).digest() code_challenge = base64.urlsafe_b64encode(digest).rstrip(b'=').decode() return code_verifier, code_challenge # Usage code_verifier, code_challenge = generate_pkce() session['pkce_verifier'] = code_verifier # Authorize URL auth_url = ( f"https://sso.gerege.mn/api/oauth/authorize?" f"client_id={CLIENT_ID}&" f"redirect_uri={quote(REDIRECT_URI)}&" f"response_type=code&" f"scope=openid+profile&" f"code_challenge={code_challenge}&" f"code_challenge_method=S256&" f"state={generate_state()}" )
// Go PKCE implementation package main import ( "crypto/rand" "crypto/sha256" "encoding/base64" ) func generatePKCE() (codeVerifier, codeChallenge string, err error) { // Generate code_verifier verifierBytes := make([]byte, 32) if _, err := rand.Read(verifierBytes); err != nil { return "", "", err } codeVerifier = base64.RawURLEncoding.EncodeToString(verifierBytes) // Generate code_challenge (S256) hash := sha256.Sum256([]byte(codeVerifier)) codeChallenge = base64.RawURLEncoding.EncodeToString(hash[:]) return codeVerifier, codeChallenge, nil } // Usage codeVerifier, codeChallenge, _ := generatePKCE() session.Set("pkce_verifier", codeVerifier) authURL := fmt.Sprintf( "https://sso.gerege.mn/api/oauth/authorize?"+ "client_id=%s&"+ "redirect_uri=%s&"+ "response_type=code&"+ "scope=openid+profile&"+ "code_challenge=%s&"+ "code_challenge_method=S256&"+ "state=%s", clientID, url.QueryEscape(redirectURI), codeChallenge, state, )

8. Код жишээнүүд

// Node.js/Express example with PKCE const express = require('express'); const crypto = require('crypto'); const axios = require('axios'); const app = express(); const CLIENT_ID = 'your-client-id'; const CLIENT_SECRET = 'your-client-secret'; const REDIRECT_URI = 'https://yourapp.com/callback'; // Step 1: Initiate OAuth flow app.get('/login', (req, res) => { // Generate PKCE const codeVerifier = crypto.randomBytes(32).toString('base64url'); const codeChallenge = crypto.createHash('sha256') .update(codeVerifier).digest('base64url'); // Store verifier in session req.session.codeVerifier = codeVerifier; req.session.state = crypto.randomBytes(16).toString('hex'); const authUrl = new URL('https://sso.gerege.mn/api/oauth/authorize'); authUrl.searchParams.set('client_id', CLIENT_ID); authUrl.searchParams.set('redirect_uri', REDIRECT_URI); authUrl.searchParams.set('response_type', 'code'); authUrl.searchParams.set('scope', 'openid profile'); authUrl.searchParams.set('state', req.session.state); authUrl.searchParams.set('code_challenge', codeChallenge); authUrl.searchParams.set('code_challenge_method', 'S256'); res.redirect(authUrl.toString()); }); // Step 2: Handle callback app.get('/callback', async (req, res) => { const { code, state } = req.query; // Verify state if (state !== req.session.state) { return res.status(400).send('Invalid state'); } // Exchange code for token const tokenResponse = await axios.post( 'https://sso.gerege.mn/api/oauth/token', new URLSearchParams({ grant_type: 'authorization_code', code, redirect_uri: REDIRECT_URI, client_id: CLIENT_ID, client_secret: CLIENT_SECRET, code_verifier: req.session.codeVerifier, }), { headers: { 'Content-Type': 'application/x-www-form-urlencoded' } } ); const { access_token } = tokenResponse.data; // Decode JWT to get user info const payload = JSON.parse( Buffer.from(access_token.split('.')[1], 'base64').toString() ); console.log('User:', payload.gerege); res.json({ user: payload.gerege }); });
# Python/Flask example with PKCE from flask import Flask, redirect, request, session, url_for import secrets import hashlib import base64 import requests import jwt app = Flask(__name__) app.secret_key = 'your-secret-key' CLIENT_ID = 'your-client-id' CLIENT_SECRET = 'your-client-secret' REDIRECT_URI = 'https://yourapp.com/callback' @app.route('/login') def login(): # Generate PKCE code_verifier = secrets.token_urlsafe(32) code_challenge = base64.urlsafe_b64encode( hashlib.sha256(code_verifier.encode()).digest() ).rstrip(b'=').decode() # Store in session session['code_verifier'] = code_verifier session['state'] = secrets.token_hex(16) auth_url = ( f"https://sso.gerege.mn/api/oauth/authorize?" f"client_id={CLIENT_ID}&" f"redirect_uri={REDIRECT_URI}&" f"response_type=code&" f"scope=openid+profile&" f"state={session['state']}&" f"code_challenge={code_challenge}&" f"code_challenge_method=S256" ) return redirect(auth_url) @app.route('/callback') def callback(): code = request.args.get('code') state = request.args.get('state') # Verify state if state != session.get('state'): return 'Invalid state', 400 # Exchange code for token token_response = requests.post( 'https://sso.gerege.mn/api/oauth/token', data={ 'grant_type': 'authorization_code', 'code': code, 'redirect_uri': REDIRECT_URI, 'client_id': CLIENT_ID, 'client_secret': CLIENT_SECRET, 'code_verifier': session['code_verifier'], } ) access_token = token_response.json()['access_token'] # Decode JWT (without verification for demo) payload = jwt.decode(access_token, options={"verify_signature": False}) return { 'user': payload['gerege'], 'email': payload['email'] }
